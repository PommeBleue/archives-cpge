	\subsubsection*{\textsc{Pierre-Gabriel Berlureau} \Stars{0.5}}
	Soient $(G, \cdot)$ un groupe et $I$ un ensemble non vide. On considère une famille de sous-groupes de $G$, notée $(G_i)_{i\in I}$, telle que \[\forall (i,j)\in I^2,\ \exists k\in I,\ G_i\cup G_j\subset G_k\]
	Montrer que $\bigcup_{i\in I}G_i$ est un sous-groupe de $G$.
	\subsubsection*{\textsc{Matteo Delfour} \Stars{2}}
		Soit $(E, \cdot)$ un magma associatif tel qu'il existe $n\in\N$, supérieur ou égal à 2, tel que \[\forall(x,y)\in E^2,\ (xy)^n=yx\]
		Montrer que $\cdot$ est commutative.
	\subsubsection*{\textsc{Yanis Grigy} \Stars{3}}
		Montrer que pour tout $n\in\N^*$, $\ln(n!)\geq n\ln\frac ne$ et en déduire que \[\forall (x_1,\dots,x_n)\in(\Rpe)^n,\quad \left(\prod_{k=1}^nx_k\right)^{\frac1n}\leq \frac e{n^2}\sum_{k=1}^nkx_k\]
	\subsubsection*{\textsc{Louis Marchal} \Stars{4}}
		Soient $a<b\in\R$ et $f:[a,b]\to\R$ une application bornée vérifiant \[\forall(x,y)\in[a,b]^2,\quad f\left(\frac{x+y}2\right)\leq\frac{f(x)+f(y)}2\]
		Montrer que $f$ est convexe sur $[a,b]$.
